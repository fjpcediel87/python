def adjacency_list_to_matrix(adj_list):
    n = len(adj_list)
    matrix = [[0] * n for _ in range(n)]
    for i, nodes in adj_list.items():        
        for j in nodes:
            matrix[i][j] = 1
        print(matrix[i][:])
    return matrix
